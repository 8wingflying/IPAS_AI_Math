import matplotlib.pyplot as plt
import numpy as np

# Generate normal distribution plot
x = np.linspace(-4, 4, 400)
mu = 0
sigma = 1
y = 1/(np.sqrt(2*np.pi)*sigma) * np.exp(-(x-mu)**2/(2*sigma**2))

plt.figure()
plt.plot(x, y)
plt.xlabel("x")
plt.ylabel("PDF")
plt.title("Standard Normal Distribution")
plt.grid(True)
plt.savefig("/mnt/data/normal_distribution.png")
plt.close()
```

import sympy as sp

x = sp.Symbol('x', real=True)
mu = sp.Symbol('mu', real=True)
sigma = sp.Symbol('sigma', positive=True)

pdf = 1/(sp.sqrt(2*sp.pi)*sigma) * sp.exp(-(x-mu)**2 / (2*sigma**2))

EX = sp.integrate(x * pdf, (x, -sp.oo, sp.oo))
EX2 = sp.integrate(x**2 * pdf, (x, -sp.oo, sp.oo))

VarX = sp.simplify(EX2 - EX**2)

EX, VarX
```
