- CDF
```python
import matplotlib.pyplot as plt
from scipy.stats import norm
import numpy as np

# Reuse the x-values from the previous plot
x = np.linspace(-4, 4, 400)

# Calculate the CDF using scipy.stats.norm.cdf
# For a standard normal distribution (mu=0, sigma=1)
cdf_values = norm.cdf(x, loc=0, scale=1)

# ç•«åœ–
plt.figure()
plt.plot(x, cdf_values)
plt.xlabel("x")
plt.ylabel("CDF")
plt.title("Standard Normal Distribution CDF")
plt.grid(True)
plt.show()
```

```python
import sympy as sp

x = sp.Symbol('x', real=True)
mu = sp.Symbol('mu', real=True)
sigma = sp.Symbol('sigma', positive=True)

pdf = 1/(sp.sqrt(2*sp.pi)*sigma) * sp.exp(-(x-mu)**2 / (2*sigma**2))

EX = sp.integrate(x * pdf, (x, -sp.oo, sp.oo))
EX2 = sp.integrate(x**2 * pdf, (x, -sp.oo, sp.oo))

VarX = sp.simplify(EX2 - EX**2)

EX, VarX
```
